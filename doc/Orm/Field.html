		Field
		
			__construct
		
					(string $fieldname, CAST $cast, [int $size = null], [true $nullable = false], [KEY $KEY = null], [string $KEYName = null])
			

	
			
    
					
  • 
				string
				$fieldname: the (unique) name of the field			
    • 
					
  • 
				CAST
				$cast: The CAST value of the field example : CAST::$INTEGER			
    • 
					
  • 
				int
				$size: The max size of the field. May be null if $type is Date, Time, Buffer, None ...			
    • 
					
  • 
				true
				$nullable: if the value may be NULL. Default value is false			
    • 
					
  • 
				KEY
				$KEY: the KEY value of the field. May be null example: KEY::$PK for a primary key			
    • 
					
  • 
				string
				$KEYName: the name of the key. Only used for ForeignKey and AssociateKey example : "Customer.customer_id" in the field "customer" of an entity "Order".
